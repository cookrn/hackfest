(function( window ){

  var HACKFEST = {

    init: function () {

      // Begin your mad JavaScript in here!
      HACKFEST.bindEvents();

    },

    bindEvents: function () {

      $('#zombie_whale').bind( 'click' , function(e){
        HACKFEST.helpers.moveTo( '#zombie_whale', '#hipster_octopus', function(){
          HACKFEST.form.fadeIn(800, function(){
            HACKFEST.to.val('#big_squid');
            HACKFEST.message.val('Yo dawg you have AIDS.');
            HACKFEST.form.trigger('submit');
          });
        });
      } );

      HACKFEST.form.bind( 'submit' , function(e){
        e.preventDefault();
        HACKFEST.helpers.sendMessage();
      } );

    },

    helpers: {

      moveTo: function moveTo( animal , destination , callback ){
        $(animal).animate($(destination).position(),2000,callback);
      },
      
      sendMessage: function sendMessage( callback ){        
        HACKFEST.form.fadeOut( 800 , function(){
          var destination = HACKFEST.to.val();
          HACKFEST.helpers.moveTo( '#zombie_whale' , destination , function(){ 
            HACKFEST.form.css( $(destination).position() );
            HACKFEST.form.fadeIn(800, function(){
              HACKFEST.to.val('');
              HACKFEST.message.val('');              
            });
          } );
        } );
      },
      

    },

    friends: [ "#big_squid" , "#star_starfish" , "#drunk_clam" ],
    form: $('#message_form'),
    to: $('#to'),
    message: $('#message')

  };

  $( document ).ready( HACKFEST.init );

  })( this );
